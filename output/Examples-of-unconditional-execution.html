Examples of unconditional execution

;; Print numbers and their squares.
;; The DO construct applies to multiple forms.
 (loop for i from 1 to 3
       do (print i)
          (print (* i i)))
 |>  1 
 |>  1 
 |>  2 
 |>  4 
 |>  3 
 |>  9
